def initialize(context):
    # 设置基准为沪深300
    set_benchmark('000300.SS')
    # 初始化变量
    g.security = []
    g.max_hold_num = 5
    # 买入条件参数
    g.buy_conditions = {
        'rsi6': 50,
        'volume_ratio': 2
    }
    # 卖出条件参数
    g.sell_conditions = {
        'max_drawdown': 0.05,
        'cost_drawdown': 0.05
    }
    # 初始化持仓
    g.positions = []
    # 记录每只持仓股票的最高价（Ptrade不提供，需手动维护）
    g.max_prices = {}
    # MACD需要至少26+9=35根K线，取40留余量（原代码只取20根，MACD永远返回None）
    g.history_bars = 40


def before_trading_start(context, data):
    g.current_date = context.blotter.current_dt.strftime("%Y%m%d")
    # 获取全部A股
    g.security = get_Ashares(g.current_date)
    # 批量过滤ST股票（1次API调用，替代逐个循环5000+次）
    status_dict = get_stock_status(g.security, 'ST')
    g.filtered_stocks = [s for s in g.security if not status_dict.get(s)]


def handle_data(context, data):
    # 获取当前持仓（返回 {code: Position对象}）
    current_positions = get_positions()

    # ========== 卖出逻辑 ==========
    for stock in list(g.positions):
        # 持仓已被清空（之前卖出未从列表移除）
        if stock not in current_positions:
            g.positions.remove(stock)
            g.max_prices.pop(stock, None)
            continue

        pos = current_positions[stock]
        current_price = pos.last_sale_price
        cost_price = pos.cost_basis

        # 更新持仓最高价
        g.max_prices[stock] = max(g.max_prices.get(stock, current_price), current_price)
        max_price = g.max_prices[stock]

        # 止损检查
        if (current_price / max_price - 1 < -g.sell_conditions['max_drawdown'] or
                (cost_price > 0 and current_price / cost_price - 1 < -g.sell_conditions['cost_drawdown'])):
            order_target(stock, 0)
            g.positions.remove(stock)
            g.max_prices.pop(stock, None)

    # ========== 买入逻辑 ==========
    if len(g.positions) >= g.max_hold_num:
        return

    need = g.max_hold_num - len(g.positions)
    candidates = []

    # 排除已持仓的股票
    stocks_to_check = [s for s in g.filtered_stocks if s not in g.positions]
    if not stocks_to_check:
        return

    # 批量获取历史数据（1次API调用，替代逐个循环5000+次）
    try:
        history = get_history(g.history_bars, '1d', ['close', 'volume'], stocks_to_check)
    except Exception:
        return

    for stock in stocks_to_check:
        if len(candidates) >= need:
            break

        try:
            # get_history 多股票返回 dict{code: DataFrame}
            stock_df = history[stock]
            closes = stock_df['close']
            volumes = stock_df['volume']

            # 数据不足则跳过（MACD需要26根以上）
            if len(closes.dropna()) < 30:
                continue

            # 计算技术指标
            rsi6 = calculate_rsi(closes, 6)
            macd_result = calculate_macd(closes)
            boll = calculate_boll(closes)

            if rsi6 is None or macd_result is None or boll is None:
                continue

            rsi6_value = rsi6.iloc[-1]
            macd_value = macd_result['macd'].iloc[-1]
            macd_signal = macd_result['signal'].iloc[-1]
            boll_close = closes.iloc[-1]
            boll_mid = boll['mid'].iloc[-1]
            avg_vol = volumes.mean()
            volume_ratio = volumes.iloc[-1] / avg_vol if avg_vol > 0 else 0

            # 买入条件：RSI6>50 且 MACD>信号线 且 价格>布林中轨 且 放量2倍
            if (rsi6_value > g.buy_conditions['rsi6'] and
                    macd_value > macd_signal and
                    boll_close > boll_mid and
                    volume_ratio > g.buy_conditions['volume_ratio']):
                candidates.append(stock)
        except Exception:
            continue

    # 执行买入
    for stock in candidates[:need]:
        order_target(stock, 1000)
        g.positions.append(stock)
        g.max_prices[stock] = 0


def calculate_rsi(prices, period):
    if len(prices) < period:
        return None
    delta = prices.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
    rs = gain / loss
    return 100 - (100 / (1 + rs))


def calculate_macd(prices, short_window=12, long_window=26, signal_window=9):
    if len(prices) < long_window:
        return None
    short_ema = prices.ewm(span=short_window, adjust=False).mean()
    long_ema = prices.ewm(span=long_window, adjust=False).mean()
    macd = short_ema - long_ema
    signal = macd.ewm(span=signal_window, adjust=False).mean()
    return {'macd': macd, 'signal': signal}


def calculate_boll(prices, window=20):
    if len(prices) < window:
        return None
    mid = prices.rolling(window=window).mean()
    std = prices.rolling(window=window).std()
    upper = mid + 2 * std
    lower = mid - 2 * std
    return {'mid': mid, 'upper': upper, 'lower': lower}
